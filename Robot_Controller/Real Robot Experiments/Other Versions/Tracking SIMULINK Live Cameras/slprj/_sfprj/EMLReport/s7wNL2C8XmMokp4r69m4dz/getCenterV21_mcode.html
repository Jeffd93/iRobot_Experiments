1function [L_p, R_p, sWindowOut] = getCenterV2(leftImage, rightImage, sWindowIn)
 2
 3    %#codegen
 4    L_p=zeros(2,4);
 5    R_p=zeros(2,4);
 6    
 7    m = size(leftImage,1);
 8    sWindowOut = zeros(size(sWindowIn));
 9    
10    lwx = zeros(1,100);
11    lwy = zeros(1,100);
12    coder.varsize('lwx', [1 128]);
13    coder.varsize('lwy', [1 128]);
14    
15    for i=1:4
16        lw = sWindowIn(i,:);
17        lwx = lw(1):lw(1)+lw(3);
18        lwy = lw(2):lw(2)+lw(4);
19        
20        rw = sWindowIn(i+4,:);
21        rwx = 1:10;%rw(1):rw(1)+rw(3);
22        rwy = 1:10;%rw(2):rw(2)+rw(4);
23        
24        pWinL = leftImage(lwy,lwx,:);
25        pWinR = rightImage(rwy,rwx,:);
26                
27        pWinL = imcomplement(pWinL);        
28        pWinR = imcomplement(pWinR);        
29    
30        pGrayL = 0.2989 * pWinL(:,:,1) + 0.5870 * pWinL(:,:,2) + 0.1140 * pWinL(:,:,3);
31        pGrayR = 0.2989 * pWinR(:,:,1) + 0.5870 * pWinR(:,:,2) + 0.1140 * pWinR(:,:,3);
32    
33        pWinLF = pWinL(:,:,3)-pGrayL;       % imsubtract(rgb1(:,:,3),gray)
34        pWinRF = pWinR(:,:,3)-pGrayR;       % imsubtract(rgb1(:,:,3),gray) 
35
36%         pWinLF = medfilt2(pWinLF);     % medfilt2(rgb2,[3 3]);
37%         pWinRF = medfilt2(pWinRF);     % medfilt2(rgb2,[3 3]);
38% 
39%         pWinLF = mat2gray(pWinLF);
40%         pWinRF = mat2gray(pWinRF);
41
42        pBWL = pWinLF > 0.2;
43        pBWR = pWinRF > 0.2;
44
45        %finds yellow regions in image using BlobAnalysis.
46        hblob = vision.BlobAnalysis;
47
48        [~,cL,bbL] = step(hblob, pBWL); % get [x y] is a centroid of region
49        [~,cR,bbR] = step(hblob, pBWR); % get [x y] is a centroid of region
50        
51        L_p(:,i) = cL(1,:) + lw(1:2);
52        R_p(:,i) = cR(1,:) + rw(1:2);
53        
54        coeffL = double([-0.3 0;0 -0.3;0.3 0;0 0.3]); szL = double(bbL(3:4)');
55        coeffR = double([-0.3 0;0 -0.3;0.3 0;0 0.3]); szR = double(bbR(3:4)');
56        
57        sWindowOut(i,:)   = round([lw(1); lw(2); 0; 0] + double(bbL') + coeffL*szL)';
58        sWindowOut(i+4,:) = round([rw(1); rw(2); 0; 0] + double(bbR') + coeffR*szR)';
59    end
60    
61    L_p(2,:) = m - L_p(2,:);
62    R_p(2,:) = m - R_p(2,:);
63end
