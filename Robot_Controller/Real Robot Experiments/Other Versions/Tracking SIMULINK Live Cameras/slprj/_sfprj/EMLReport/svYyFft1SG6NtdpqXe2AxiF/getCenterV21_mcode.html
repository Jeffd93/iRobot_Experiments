1function [L_p, R_p, sWindowOut] = getCenterV2(leftImage, rightImage, sWindowIn)
 2
 3    %#codegen
 4    L_p=zeros(2,4);
 5    R_p=zeros(2,4);
 6    
 7    m = size(leftImage,1);
 8    sWindowOut = zeros(size(sWindowIn));
 9    
10
11    
12    for i=1:4
13        lw = sWindowIn(i,:);
14        
15            coder.varsize('lwx');
16    coder.varsize('lwy');
17        
18        lwx = lw(1):lw(1)+lw(3);
19        lwy = lw(2):lw(2)+lw(4);
20        
21        rw = sWindowIn(i+4,:);
22        rwx = rw(1):rw(1)+rw(3);
23        rwy = rw(2):rw(2)+rw(4);
24        
25        pWinL = leftImage(lwy,lwx,:);
26        pWinR = rightImage(rwy,rwx,:);
27                
28        pWinL = imcomplement(pWinL);        
29        pWinR = imcomplement(pWinR);        
30    
31        pGrayL = 0.2989 * pWinL(:,:,1) + 0.5870 * pWinL(:,:,2) + 0.1140 * pWinL(:,:,3);
32        pGrayR = 0.2989 * pWinR(:,:,1) + 0.5870 * pWinR(:,:,2) + 0.1140 * pWinR(:,:,3);
33    
34        pWinLF = pWinL(:,:,3)-pGrayL;       % imsubtract(rgb1(:,:,3),gray)
35        pWinRF = pWinR(:,:,3)-pGrayR;       % imsubtract(rgb1(:,:,3),gray) 
36
37%         pWinLF = medfilt2(pWinLF);     % medfilt2(rgb2,[3 3]);
38%         pWinRF = medfilt2(pWinRF);     % medfilt2(rgb2,[3 3]);
39% 
40%         pWinLF = mat2gray(pWinLF);
41%         pWinRF = mat2gray(pWinRF);
42
43        pBWL = pWinLF > 0.2;
44        pBWR = pWinRF > 0.2;
45
46        %finds yellow regions in image using BlobAnalysis.
47        hblob = vision.BlobAnalysis;
48
49        [~,cL,bbL] = step(hblob, pBWL); % get [x y] is a centroid of region
50        [~,cR,bbR] = step(hblob, pBWR); % get [x y] is a centroid of region
51        
52        L_p(:,i) = cL(1,:) + lw(1:2);
53        R_p(:,i) = cR(1,:) + rw(1:2);
54        
55        coeffL = double([-0.3 0;0 -0.3;0.3 0;0 0.3]); szL = double(bbL(3:4)');
56        coeffR = double([-0.3 0;0 -0.3;0.3 0;0 0.3]); szR = double(bbR(3:4)');
57        
58        sWindowOut(i,:)   = round([lw(1); lw(2); 0; 0] + double(bbL') + coeffL*szL)';
59        sWindowOut(i+4,:) = round([rw(1); rw(2); 0; 0] + double(bbR') + coeffR*szR)';
60    end
61    
62    L_p(2,:) = m - L_p(2,:);
63    R_p(2,:) = m - R_p(2,:);
64end
