1function [L_p, R_p] = getCenter(l_rgb,r_rgb)
 2
 3    %#codegen
 4    L_p=zeros(2,4);
 5    R_p=zeros(2,4);
 6    
 7    %input left image(l_rgb), right image(r_rgb), Pose target position 
 8    [m,n,c]=size(l_rgb);
 9    
10    %define image variables for processing
11    l_gray=zeros(m,n);      r_gray=zeros(m,n);
12    l_rgb1=zeros(m,n,c);    r_rgb1=zeros(m,n,c);
13    l_rgb2=zeros(m,n);      r_rgb2=zeros(m,n);
14
15    l_bw=false(m,n);        r_bw=false(m,n);
16
17    l_rgb1 = imcomplement(l_rgb);        % converts RGB to CMY color space 
18    r_rgb1 = imcomplement(r_rgb);        % converts RGB to CMY color space 
19    
20    l_gray = 0.2989 * l_rgb1(:,:,1) + 0.5870 * l_rgb1(:,:,2) + 0.1140 * l_rgb1(:,:,3); % rgb2gray(rgb) ;convert rgb to grayscale 
21    r_gray = 0.2989 * r_rgb1(:,:,1) + 0.5870 * r_rgb1(:,:,2) + 0.1140 * r_rgb1(:,:,3); % rgb2gray(rgb) ;convert rgb to grayscale
22    
23    l_rgb2 = l_rgb1(:,:,3)-l_gray;       % imsubtract(rgb1(:,:,3),gray)
24    r_rgb2 = r_rgb1(:,:,3)-r_gray;       % imsubtract(rgb1(:,:,3),gray) 
25    
26    l_rgb2 = medfilt2(l_rgb2);     % medfilt2(rgb2,[3 3]);
27    r_rgb2 = medfilt2(r_rgb2);     % medfilt2(rgb2,[3 3]);
28    
29    l_rgb2 = myMat2gray(l_rgb2);
30    r_rgb2 = myMat2gray(r_rgb2);
31    
32    l_bw = l_rgb2 > 0.6;
33    r_bw = r_rgb2 > 0.6;
34
35    %finds yellow regions in image using BlobAnalysis.
36    hblob = vision.BlobAnalysis;
37
38    [~,l_center,~] = step(hblob, l_bw); % get [x y] is a centroid of region
39    [~,r_center,~] = step(hblob, r_bw); % get [x y] is a centroid of region
40    
41    %Send Left Camera Points
42    ml = size(l_center,1);
43    if ml > 4
44        ml = 4;
45    end
46    for i=1:ml
47        L_p(:,i)=l_center(i,:);
48    end
49    
50    %Send Right Camera Points
51    mr = size(r_center,1);
52    if mr > 4
53        mr = 4;
54    end
55    for i=1:mr
56        R_p(:,i)=r_center(i,:);
57    end
58    
59    L_p = orderPoints(L_p); 
60    R_p = orderPoints(R_p);
61    
62    
63%     coder.extrinsic('imshow','plot');
64%     imshow(l_rgb2);hold on;
65%     plot(L_p(1,:),L_p(2,:),'ro');
66    
67    L_p(2,:) = m - L_p(2,:);
68    R_p(2,:) = m - R_p(2,:);
69end
